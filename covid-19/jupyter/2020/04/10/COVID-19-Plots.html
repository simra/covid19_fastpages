vals_1k = tidy[tidy['Confirmed'].apply(lambda x: x>1000)].groupby('Country/Region').apply(lambda df:pd.Series({'Confirmed':df['Confirmed'].min()}))

vals_1k

shifted['Confirmed']=shifted['Confirmed_x']

shifted = shifted[shifted['date']>=shifted['date_1k']]
shifted['days'] = shifted['date'] - shifted['date_1k']
shifted['days'] = shifted['days'].apply(lambda x: x.days)
